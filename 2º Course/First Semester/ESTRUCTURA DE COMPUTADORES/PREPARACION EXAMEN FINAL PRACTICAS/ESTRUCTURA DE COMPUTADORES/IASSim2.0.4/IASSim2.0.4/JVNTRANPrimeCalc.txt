/*
    Fills the PRIMES array with the first
    10 prime numbers
*/
CONSTANT INTEGER NUM_PRIMES = 10
INTEGER PRIMES(NUM_PRIMES)
PRIMES(1) = 2

//FIND THE OTHER 9 PRIMES
INTEGER INDEX = 2 //INDEX INTO PRIMES
INTEGER NEXTINT = 3 //NEXT INT TO TEST FOR PRIMALITY
INTEGER J //LOOP VARIABLE
DO WHILE (INDEX <= 10)
    J = 1
    DO WHILE (J < INDEX)
        IF (NEXTINT % PRIMES(J) == 0)
            BREAK
        END IF
        J = J + 1
    END DO
    IF (J == INDEX)
        PRIMES(INDEX) = NEXTINT
	INDEX = INDEX + 1
    END IF
    NEXTINT = NEXTINT + 2
END DO

HALT
